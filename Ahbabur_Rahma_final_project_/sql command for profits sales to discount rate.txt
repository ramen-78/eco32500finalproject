SELECT 
    DATEPART(month, od.order_date_DateOrders) AS Order_Month,
    COUNT(*) AS Total_Orders,
    AVG(od.Sales) AS Avg_Sales,
    AVG(od.Benefit_per_order) AS Avg_Benefit_per_order,
    AVG(od.Order_Item_Discount_Rate) AS Avg_Discount_Rate,
    AVG(CAST(od.Order_Profit_Per_Order AS DECIMAL(18, 2))) AS Avg_Profit
FROM 
    OrderData od
WHERE
    od.Order_Status = 'COMPLETE'
    AND YEAR(od.order_date_DateOrders) != 2018
GROUP BY 
    DATEPART(month, od.order_date_DateOrders)
ORDER BY 
    Order_Month;
